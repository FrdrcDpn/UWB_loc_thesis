46 void keyboard_callback(unsigned char key, __attribute__((unused)) int a, __attribute__((unused)) int b)
47 {
48  switch (key) {
49  case 'c': // Center the animation
50  terminalinfo::info_msg("Recentering Animation.");
51  center_x = 0;
52  center_y = 0;
53  break;
54  case 'z': // Reset the zoom to the default value
55  terminalinfo::info_msg("Resetting zoom.");
56  zoom = param->zoom();
57  break;
58  case 'q': // End the simulation and quit
59  terminalinfo::info_msg("Quitting Swarmulator.");
60  program_running = false;
61  break;
62  case 'p': // Pause the simulation
63  if (!paused) {
64  terminalinfo::info_msg("Paused. Press `r' to resume or `s' to step forward.");
65  paused = true;
66  mtx.lock();
67  }
68  break;
69  case 'r': // Resume the simulation (if paused)
70  if (paused) {
71  terminalinfo::info_msg("Resuming.");
72  paused = false;
73  mtx.unlock();
74  }
75  break;
76  case 's': // Step through the simulation. Very useful for debugging or analyzing what's going on.
77  if (paused) {
78  terminalinfo::info_msg("Stepping through. Press `s' to keep stepping forwrad to `r' to resume. ");
79  mtx.unlock();
80  int t_wait = (int)1e6 * (1.0 / (param->simulation_updatefreq() * param->simulation_realtimefactor()));
81  std::this_thread::sleep_for(std::chrono::microseconds(t_wait));
82  mtx.lock();
83  }
84  break;
85  case 'a': // Draw and simulate a new agent, initialized at the current pointer position
86  if (!paused) {
87  terminalinfo::info_msg("Drawing new agent.");
88  random_generator rg;
89  std::vector<float> states = {pointer_y, pointer_x, 0.0, 0.0, 0.0, 0.0, rg.uniform_float(-M_PI, M_PI), 0.0}; // Initial positions/states
90  create_new_agent(s.size(), states);
91  break;
92  }
93  case 'm': // Toggle the real time parameter between 1 and default, so as to better understand what's going on
94  terminalinfo::info_msg("Toggle realtime factor between 1 and the specified value.");
95  if (param->simulation_realtimefactor() != 1) {
96  realtimefactor = param->simulation_realtimefactor();
97  param->simulation_realtimefactor() = 1;
98  } else {
99  param->simulation_realtimefactor() = realtimefactor;
100  }
101  break;
102  case '1': // Turn the agent with 0.1 rad/s
103  s[0]->manualpsi_delta = 0.1;
104  break;
105  case '2': // Turn the agent with -0.1 rad/s
106  s[0]->manualpsi_delta = -0.1;
107  break;
108  case 'n': // Quit and restart swarmulator
109  terminalinfo::info_msg("Restarting.");
110  std::stringstream ss;
111  ss << "pkill swarmulator && ./swarmulator " << nagents;
112  system(ss.str().c_str());
113  break;
114  }
115 }

178  // End position of the new wall

180  float wall_y_1 = -((float)y / (float)glutGet(GLUT_WINDOW_HEIGHT) * 8. - 4.) / (zoom_scale * yrat) - center_y;
181  // Generate new wall
182  environment.add_wall(wall_x_0, wall_y_0, wall_x_1, wall_y_1);
183  }
184 
185  // Zoom wheel
186  if (button == GLUT_WHEEL_UP) {
187  zoom += param->mouse_zoom_speed();
188  } else if (button == GLUT_WHEEL_DOWN) {
189  zoom -= param->mouse_zoom_speed();
190  }
191 
192  // Guard on too much / too little zoom
